114 void
115 MmWaveVehicularSpectrumPhyTestCase1::StartTest (double dist)
116 {
117  // This test creates two MmWaveSidelinkSpectrumPhy instances (tx and rx)
118  // and connects them through a SpectrumChannel. The SpectrumChannel is
119  // configured with a ConstantSpeedPropagationDelayModel and a
120  // FriisSpectrumPropagationLossModel.
121  // The tx instance sends a dummy signal to the rx, which computes the SNR.
122  // The computed SNR is compared with the expected SNR, which is computed
123  // offline using the link budget and the Friis formulas.
124 
125  double distance = dist; // distance between tx and rx
126 
127  // create the mobility model
128  Ptr<MobilityModel> tx_mm = CreateObject<ConstantPositionMobilityModel> ();
129  Ptr<MobilityModel> rx_mm = CreateObject<ConstantPositionMobilityModel> ();
130  tx_mm->SetPosition (Vector (0.0, 0.0, 0.0));
131  rx_mm->SetPosition (Vector (distance, 0.0, 0.0));
132 
133  // create the antenna
134  Ptr<AntennaModel> tx_am = CreateObject<IsotropicAntennaModel> ();
135  Ptr<AntennaModel> rx_am = CreateObject<IsotropicAntennaModel> ();
136 
137  // create the channel
138  SpectrumChannelHelper sh = SpectrumChannelHelper::Default ();
139  Ptr<SpectrumChannel> sc = sh.Create ();
140 
141  // create and configure the tx spectrum phy
142  Ptr<MmWaveSidelinkSpectrumPhy> tx_ssp = CreateObject<MmWaveSidelinkSpectrumPhy> ();
143  tx_ssp->SetMobility (tx_mm);
144  tx_ssp->SetAntenna (tx_am);
145  tx_ssp->SetChannel (sc);
146 
147  // create and configure the tx spectrum phy
148  Ptr<MmWaveSidelinkSpectrumPhy> rx_ssp = CreateObject<MmWaveSidelinkSpectrumPhy> ();
149  rx_ssp->SetMobility (rx_mm);
150  rx_ssp->SetAntenna (rx_am);
151  rx_ssp->SetChannel (sc);
152 
153  // add the rx spectrum phy instance to the spectrum channel
154  sc->AddRx (rx_ssp);
155 
156  // connect the rx callback to the sink
157  rx_ssp->SetPhyRxDataEndOkCallback (MakeCallback (&MmWaveVehicularSpectrumPhyTestCase1::Rx, this));
158 
159  // create and configure the chunk processor
160  Ptr<mmwave::mmWaveChunkProcessor> pData = Create<mmwave::mmWaveChunkProcessor> ();
161  pData->AddCallback (MakeCallback (&MmWaveSidelinkSpectrumPhy::UpdateSinrPerceived, rx_ssp));
162  pData->AddCallback (MakeCallback (&MmWaveVehicularSpectrumPhyTestCase1::UpdateSinrPerceived, this));
163  rx_ssp->AddDataSinrChunkProcessor (pData);
164 
165  // create the tx psd
166  Ptr<mmwave::MmWavePhyMacCommon> pmc = CreateObject<mmwave::MmWavePhyMacCommon> ();
167  double txp = 30.0; // transmission power in dBm
168  std::vector<int> subChannelsForTx (pmc->GetNumChunks ());
169 
170  // create the transmission mask, use all the available subchannels
171  uint32_t i = 0;
172  for (auto& sc : subChannelsForTx)
173  {
174  sc = i++;
175  }
176 
177  Ptr<SpectrumValue> txPsd = mmwave::MmWaveSpectrumValueHelper::CreateTxPowerSpectralDensity (pmc, txp, subChannelsForTx);
178  tx_ssp->SetTxPowerSpectralDensity (txPsd);
179 
180 
181  // set the rx noise psd
182  double noiseFigure = 5.0; // noise figure in dB
183  Ptr<SpectrumValue> noisePsd = mmwave::MmWaveSpectrumValueHelper::CreateNoisePowerSpectralDensity (pmc, noiseFigure);
184  rx_ssp->SetNoisePowerSpectralDensity (noisePsd);
185 
186  // set the device for the receiving side, as the MmWaveSidelinkSpectrumPhy checks the RNTI of the receiver
187  // to drop or not the packet
188  uint16_t rxRnti = 1;
189  Ptr<MmWaveSidelinkPhy> phy = CreateObject<MmWaveSidelinkPhy> (rx_ssp, pmc);
190  Ptr<MmWaveSidelinkMac> mac = CreateObject<MmWaveSidelinkMac> (pmc);
191  phy->SetPhySapUser (mac->GetPhySapUser ());
192  mac->SetPhySapProvider (phy->GetPhySapProvider ());
193  mac->SetRnti (rxRnti);
194  Ptr<MmWaveVehicularNetDevice> device = CreateObject<MmWaveVehicularNetDevice> (phy, mac);
195  NodeContainer nc;
196  nc.Create(1);
197  rx_ssp->SetDevice (device);
198  device->SetNode(nc.Get(0));
199  nc.Get(0)->AddDevice(device);
200 
201  // send an empty packet burst
202  Ptr<Packet> p = Create<Packet> (20);
203  Ptr<PacketBurst> pb = CreateObject<PacketBurst> ();
204  pb->AddPacket (p);
205  Time duration = MilliSeconds (1); // packet duration
206  uint8_t mcs = 0; // MCS
207  uint8_t numSym = 14; // number of symbols dedicated to the transport block
208  uint8_t size = 20; // size of the transport block
209 
210  // send the transport block through the spectrum channel
211  tx_ssp->StartTxDataFrames (pb, duration, mcs, size, numSym, 0, rxRnti, subChannelsForTx);
212 
213  // compute the expected SINR
214  m_expectedSinr = txp + 20 * log10 (3e8 / (4 * M_PI * distance * pmc->GetCenterFrequency ())) + 114 - noiseFigure - 10 * log10 (pmc->GetBandwidth () / 1e6);
215 
216  Simulator::Stop (MilliSeconds(2));
217  Simulator::Run ();
218  Simulator::Destroy ();
219 
220 }
