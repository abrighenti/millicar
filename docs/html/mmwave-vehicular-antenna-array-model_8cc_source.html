175 void
176 MmWaveVehicularAntennaArrayModel::SetBeamformingVectorPanelDevices (Ptr<NetDevice> thisDevice, Ptr<NetDevice> otherDevice)
177 {
178  NS_LOG_FUNCTION (this << otherDevice << Simulator::Now ());
179  m_omniTx = false;
180  complexVector_t antennaWeights;
181  int panelId = 0; // initialize all the variables
182  if (thisDevice != 0 && otherDevice != 0)
183  {
184  Vector aPos = thisDevice->GetNode ()->GetObject<MobilityModel> ()->GetPosition ();
185  NS_LOG_INFO ("aPos: " << aPos);
186  Vector bPos = otherDevice->GetNode ()->GetObject<MobilityModel> ()->GetPosition ();
187  NS_LOG_INFO ("bPos: " << bPos);
188 
189  Angles completeAngle (bPos,aPos);
190 
191  double posX = bPos.x - aPos.x;
192  double phiAngle = atan ((bPos.y - aPos.y) / posX); //horizontal angle only
193 
194  if (posX < 0)
195  {
196  phiAngle = phiAngle + M_PI;
197  }
198  if (phiAngle < 0)
199  {
200  phiAngle = phiAngle + 2 * M_PI;
201  }
202 
203  // if (m_isUe)
204  // phiAngle = M_PI;
205  // else
206  // phiAngle=0; // cast BF vectors fixed to zero [DEBUG PROCEDURE TEST LINE]
207 
208  panelId = floor (fmod (phiAngle + M_PI / m_noPlane,2 * M_PI) * m_noPlane / (2 * M_PI)); // panel id into the interval [0,N-1]
209 
210  double hAngleRadian = fmod ((phiAngle + (M_PI / m_noPlane)),2 * M_PI / m_noPlane) - (M_PI / m_noPlane);
211  double vAngleRadian = completeAngle.theta;
212  double power = 1 / sqrt (m_totNoArrayElements);
213  uint16_t antennaNum [2];
214  antennaNum[0] = sqrt (m_totNoArrayElements);
215  antennaNum[1] = sqrt (m_totNoArrayElements);
216  NS_LOG_INFO ("hAngleRadian: " << hAngleRadian);
217 
218  for (uint64_t ind = 0; ind < m_totNoArrayElements; ind++)
219  {
220  Vector loc = GetAntennaLocation (ind, antennaNum);
221  double phase = -2 * M_PI * (sin (vAngleRadian) * cos (hAngleRadian) * loc.x
222  + sin (vAngleRadian) * sin (hAngleRadian) * loc.y
223  + cos (vAngleRadian) * loc.z);
224  antennaWeights.push_back (exp (std::complex<double> (0, phase)) * power);
225  }
226 
227  std::map< Ptr<NetDevice>, std::pair<complexVector_t,int> >::iterator iter = m_beamformingVectorPanelMap.find (otherDevice);
228  if (iter != m_beamformingVectorPanelMap.end ())
229  {
230  (*iter).second = std::make_pair (antennaWeights, panelId);
231  m_lastUpdatePairMap[otherDevice] = Simulator::Now ();
232  }
233  else
234  {
235  m_beamformingVectorPanelMap.insert (std::make_pair (otherDevice, std::make_pair (antennaWeights, panelId) ));
236  m_lastUpdatePairMap.insert (std::make_pair (otherDevice, Simulator::Now ()));
237 
238  NS_LOG_INFO ("m_lastUpdatePairMap.size " << m_lastUpdatePairMap.size ());
239  }
240  }
241  m_beamformingVector = antennaWeights;
242  m_currentPanelId = panelId;
243  m_currentDev = otherDevice;
244  NS_LOG_INFO ("panelId: " << panelId);
245  NS_LOG_INFO ("otherDevice: " << otherDevice);
246 }

334 double
335 MmWaveVehicularAntennaArrayModel::GetRadiationPattern (double vAngleRadian, double hAngleRadian)
336 {
337  if (m_isotropicElement)
338  {
339  return 1;
340  }
341 
342  while (hAngleRadian >= M_PI)
343  {
344  hAngleRadian -= 2 * M_PI;
345  }
346  while (hAngleRadian < -M_PI)
347  {
348  hAngleRadian += 2 * M_PI;
349  }
350 
351  double vAngle = vAngleRadian * 180 / M_PI;
352  double hAngle = hAngleRadian * 180 / M_PI;
353  //NS_LOG_INFO(" it is " << vAngle);
354  NS_ASSERT_MSG (vAngle >= 0&&vAngle <= 180, "the vertical angle should be the range of [0,180]");
355  //NS_LOG_INFO(" it is " << hAngle);
356  NS_ASSERT_MSG (hAngle >= -180&&hAngle <= 180, "the horizontal angle should be the range of [-180,180]");
357 
358  double A_M = 0; //front-back ratio expressed in dB
359  double SLA = 0; //side-lobe level limit expressed in dB
360 
361  if (m_antennaElementPattern == "3GPP-MmWave") //front-back ratio and side-lobe level in case of standard mmWave antenna configuration
362  {
363  A_M = 30;
364  SLA = 30;
365  }
366  else if (m_antennaElementPattern == "3GPP-V2V") //front-back ratio and side-lobe level values in case of V2V antenna configuration
367  {
368  A_M = 25;
369  SLA = 25;
370  }
371  else
372  {
373  NS_FATAL_ERROR("Unknown antenna element pattern");
374  }
375 
376  double A_v = -1 * std::min (SLA,12 * pow ((vAngle - 90) / m_hpbw,2)); //TODO: check position of z-axis zero
377  double A_h = -1 * std::min (A_M,12 * pow (hAngle / m_hpbw,2));
378  double A = m_gMax - 1 * std::min (A_M,-1 * A_v - 1 * A_h);
379 
380  return sqrt (pow (10,A / 10)); //filed factor term converted to linear;
381 }

383 Vector
384 MmWaveVehicularAntennaArrayModel::GetAntennaLocation (uint16_t index, uint16_t* antennaNum)
385 {
386  //assume the left bottom corner is (0,0,0), and the rectangular antenna array is on the y-z plane.
387  Vector loc;
388  loc.x = 0;
389  loc.y = m_disH * (index % antennaNum[0]);
390  loc.z = m_disV * floor (index / antennaNum[0]);
391  return loc;

394 void
395 MmWaveVehicularAntennaArrayModel::SetSector (uint8_t sector, uint16_t *antennaNum, double elevation)
396 {
397  complexVector_t tempVector;
398  double hAngle_radian = M_PI * (double)sector / (double)antennaNum[1] - 0.5 * M_PI;
399  double vAngle_radian = elevation * M_PI / 180;
400  uint64_t size = antennaNum[0] * antennaNum[1];
401  double power = 1 / sqrt (size);
402  for (uint64_t ind = 0; ind < size; ind++)
403  {
404  Vector loc = GetAntennaLocation (ind, antennaNum);
405  double phase = -2 * M_PI * (sin (vAngle_radian) * cos (hAngle_radian) * loc.x
406  + sin (vAngle_radian) * sin (hAngle_radian) * loc.y
407  + cos (vAngle_radian) * loc.z);
408  tempVector.push_back (exp (std::complex<double> (0, phase)) * power);
409  }
410  m_beamformingVector = tempVector;
411 }
