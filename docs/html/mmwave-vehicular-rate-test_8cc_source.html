135 void
136 MmWaveVehicularRateTestCase::StartTest (uint8_t mcs)
137 {
138  // This test creates two nodes that exchange packets through a UDP application
139  // with a fixed sending rate.
140 
141  Time startTime = MilliSeconds (100);
142  Time endTime = MilliSeconds (500);
143 
144  Config::SetDefault ("ns3::MmWaveSidelinkMac::Mcs", UintegerValue (mcs));
145  Config::SetDefault ("ns3::MmWaveSidelinkMac::UseAmc", BooleanValue (false));
146  Config::SetDefault ("ns3::LteRlcTm::MaxTxBufferSize", UintegerValue (1024 * 1024 * 1024)); // we want to avoid buffer drops
147  Config::SetDefault ("ns3::MmWaveVehicularNetDevice::Mtu", UintegerValue (65535)); // set equal to the IP MTU
148 
149  // create the nodes
150  NodeContainer n;
151  n.Create (2);
152 
153  // create the mobility models
154  // NOTE: the position does not matter since we are not applying any channel
155  // model, we just set it to avoid failures
156  MobilityHelper mobility;
157  Ptr<ListPositionAllocator> positionAlloc = CreateObject<ListPositionAllocator> ();
158  positionAlloc->Add (Vector (0.0, 0.0, 0.0));
159  positionAlloc->Add (Vector (1.0, 0.0, 0.0));
160  mobility.SetPositionAllocator (positionAlloc);
161  mobility.SetMobilityModel ("ns3::ConstantPositionMobilityModel");
162  mobility.Install (n);
163 
164  // create and configure the helper
165  Ptr<MmWaveVehicularHelper> helper = CreateObject<MmWaveVehicularHelper> ();
166  helper->SetNumerology (3);
167  NetDeviceContainer devs = helper->InstallMmWaveVehicularNetDevices (n);
168 
169  // Install the TCP/IP stack in the two nodes
170 
171  InternetStackHelper internet;
172  internet.Install (n);
173 
174  Ipv4AddressHelper ipv4;
175  NS_LOG_INFO ("Assign IP Addresses.");
176  ipv4.SetBase ("10.1.1.0", "255.255.255.0");
177  Ipv4InterfaceContainer i = ipv4.Assign (devs);
178 
179  // Need to pair the devices in order to create a correspondence between transmitter and receiver
180  // and to populate the < IP addr, RNTI > map.
181  helper->PairDevices(devs);
182 
183  // Set the routing table
184  Ipv4StaticRoutingHelper ipv4RoutingHelper;
185  Ptr<Ipv4StaticRouting> staticRouting = ipv4RoutingHelper.GetStaticRouting (n.Get (0)->GetObject<Ipv4> ());
186  staticRouting->SetDefaultRoute (n.Get (1)->GetObject<Ipv4> ()->GetAddress (1, 0).GetLocal () , 2 );
187 
188  NS_LOG_DEBUG("IPv4 Address node 0: " << n.Get (0)->GetObject<Ipv4> ()->GetAddress (1, 0).GetLocal ());
189  NS_LOG_DEBUG("IPv4 Address node 1: " << n.Get (1)->GetObject<Ipv4> ()->GetAddress (1, 0).GetLocal ());
190 
191  //
192  // Create a UdpEchoServer application on node one.
193  //
194  uint16_t port = 4000;
195  UdpServerHelper server (port);
196  ApplicationContainer apps = server.Install (n.Get (1));
197  apps.Start (MilliSeconds (0));
198 
199  apps.Get(0)->TraceConnectWithoutContext ("Rx", MakeCallback (&MmWaveVehicularRateTestCase::Rx, this));
200 
201  //
202  // Create a UdpClient application to send UDP datagrams from node zero to
203  // node one.
204  //
205  Ptr<MmWaveAmc> m_amc = CreateObject <MmWaveAmc> (helper->GetConfigurationParameters());
206 
207  // The user is assigned to a single slot per subframe.
208  // Each slot has 14 OFDM symbols, hence the theoretical available rate is:
209  // bitPerSymbol * 14 / 1 ms
210  uint32_t availableBitsPerSlot = m_amc->GetTbSizeFromMcsSymbols(mcs, 14);
211  double availableRate = availableBitsPerSlot * 1e3; // bps
212 
213  // Configure the application to send a single packet per subframe, which has
214  // to occupy all the available resources in the slot
215  uint32_t headerSize = 30; // header sizes (UDP, IP, PDCP, RLC)
216  uint32_t packetSize = (availableBitsPerSlot / 8) - headerSize; // TB size - header sizes (UDP, IP, PDCP, RLC)
217  TimeValue interPacketInterval = MilliSeconds (1);
218  UdpEchoClientHelper client (n.Get (1)->GetObject<Ipv4> ()->GetAddress (1, 0).GetLocal (), port);
219  client.SetAttribute ("MaxPackets", UintegerValue (0xFFFFFFFF));
220  client.SetAttribute ("Interval", interPacketInterval);
221  client.SetAttribute ("PacketSize", UintegerValue (packetSize));
222  apps = client.Install (n.Get (0));
223  apps.Start (startTime);
224  apps.Stop (endTime);
225 
226  apps.Get(0)->TraceConnectWithoutContext ("Tx", MakeCallback (&MmWaveVehicularRateTestCase::Tx, this));
227 
228  Simulator::Stop (endTime + Seconds (1.0));
229  Simulator::Run ();
230  Simulator::Destroy ();
231 
232  std::cout << "----------- MCS :\t\t" << uint32_t(mcs) << " -----------" << std::endl;
233  std::cout << "Available Rate:\t\t" << availableRate/1e6 << " Mbps" << std::endl;
234  std::cout << "Packets size:\t\t" << packetSize << " Bytes" << std::endl;
235  std::cout << "Packets transmitted:\t" << g_txPackets << std::endl;
236  std::cout << "Packets received:\t" << g_rxPackets << std::endl;
237  std::cout << "Average Throughput:\t" << (double(g_rxPackets)*double(packetSize + headerSize)*8/double( g_lastReceived.GetSeconds() - g_firstReceived.GetSeconds()))/1e6 << " Mbps" << std::endl;
