285 void
286 MmWaveVehicularNetDevice::ActivateBearer(const uint8_t bearerId, const uint16_t destRnti, const Address& dest)
287 {
288  NS_LOG_FUNCTION(this << bearerId);
289  uint8_t lcid = bearerId; // set the LCID to be equal to the bearerId
290  // future extensions could consider a different mapping
291  // and will actually exploit the following map
292  m_bid2lcid.insert(std::make_pair(bearerId, lcid));
293 
294  NS_ASSERT_MSG(m_bearerToInfoMap.find (bearerId) == m_bearerToInfoMap.end (),
295  "There's another bearer associated to this bearerId: " << uint32_t(bearerId));
296 
297  EpcTft::PacketFilter slFilter;
298  slFilter.remoteAddress= Ipv4Address::ConvertFrom(dest);
299 
300  Ptr<Node> node = GetNode ();
301  Ptr<Ipv4> nodeIpv4 = node->GetObject<Ipv4> ();
302  int32_t interface = nodeIpv4->GetInterfaceForDevice (this);
303  Ipv4Address src = nodeIpv4->GetAddress (interface, 0).GetLocal ();
304  slFilter.localAddress= Ipv4Address::ConvertFrom(src);
305  //slFilter.direction= EpcTft::DOWNLINK;
306  slFilter.remoteMask= Ipv4Mask("255.255.255.255");
307  slFilter.localMask= Ipv4Mask("255.255.255.255");
308 
309  NS_LOG_DEBUG(this << " Add filter for " << Ipv4Address::ConvertFrom(dest));
310 
311  Ptr<EpcTft> tft = Create<EpcTft> (); // Create a new tft
312  tft->Add (slFilter); // Add the packet filter
313 
314  m_tftClassifier.Add(tft, bearerId);
315 
316  // Create RLC instance with specific RNTI and LCID
317  ObjectFactory rlcObjectFactory;
318  rlcObjectFactory.SetTypeId (GetRlcType(m_rlcType));
319  Ptr<LteRlc> rlc = rlcObjectFactory.Create ()->GetObject<LteRlc> ();
320 
321  rlc->SetLteMacSapProvider (m_mac->GetMacSapProvider());
322  rlc->SetRnti (destRnti); // this is the rnti of the destination
323  rlc->SetLcId (lcid);
324 
325  // Call to the MAC method that created the SAP for binding the MAC instance on this node to the RLC instance just created
326  m_mac->AddMacSapUser(lcid, rlc->GetLteMacSapUser());
327 
328  Ptr<LtePdcp> pdcp = CreateObject<LtePdcp> ();
329  pdcp->SetRnti (destRnti); // this is the rnti of the destination
330  pdcp->SetLcId (lcid);
331 
332  // Create the PDCP SAP that connects the PDCP instance to this NetDevice
333  LtePdcpSapUser* pdcpSapUser = new PdcpSpecificSidelinkPdcpSapUser (this);
334  pdcp->SetLtePdcpSapUser (pdcpSapUser);
335  pdcp->SetLteRlcSapProvider (rlc->GetLteRlcSapProvider ());
336  rlc->SetLteRlcSapUser (pdcp->GetLteRlcSapUser ());
337  rlc->Initialize (); // this is needed to trigger the BSR procedure if RLC SM is selected
338 
339  Ptr<SidelinkRadioBearerInfo> rbInfo = CreateObject<SidelinkRadioBearerInfo> ();
340  rbInfo->m_rlc= rlc;
341  rbInfo->m_pdcp = pdcp;
342  rbInfo->m_rnti = destRnti;
343 
344  NS_LOG_DEBUG(this << " MmWaveVehicularNetDevice::ActivateBearer() bid: " << (uint32_t)bearerId << " rnti: " << destRnti);
345 
346  // insert the tuple <lcid, pdcpSapProvider> in the map of this NetDevice, so that we are able to associate it to them later
347  m_bearerToInfoMap.insert (std::make_pair (bearerId, rbInfo));
348 }

350 void
351 MmWaveVehicularNetDevice::Receive (Ptr<Packet> p)
352 {
353  NS_LOG_FUNCTION (this << p);
354  NS_LOG_DEBUG ("Received packet at: " << Simulator::Now().GetSeconds() << "s");
355  uint8_t ipType;
356 
357  p->CopyData (&ipType, 1);
358  ipType = (ipType>>4) & 0x0f;
359 
360  if (ipType == 0x04)
361  {
362  m_rxCallback (this, p, Ipv4L3Protocol::PROT_NUMBER, Address ());
363  }
364  else if (ipType == 0x06)
365  {
366  m_rxCallback (this, p, Ipv6L3Protocol::PROT_NUMBER, Address ());
367  }
368  else
369  {
370  NS_ABORT_MSG ("MmWaveVehicularNetDevice::Receive - Unknown IP type...");
371  }
372 }

374 bool
375 MmWaveVehicularNetDevice::Send (Ptr<Packet> packet, const Address& dest, uint16_t protocolNumber)
376 {
377  NS_LOG_FUNCTION (this);
378 
379  // classify the incoming packet
380  uint32_t id = m_tftClassifier.Classify (packet, EpcTft::UPLINK, protocolNumber);
381  NS_ASSERT ((id & 0xFFFFFF00) == 0);
382  uint8_t bid = (uint8_t) (id & 0x000000FF);
383  uint8_t lcid = BidToLcid(bid);
384 
385  // get the SidelinkRadioBearerInfo
386  NS_ASSERT_MSG(m_bearerToInfoMap.find (bid) != m_bearerToInfoMap.end (), "No logical channel associated to this communication");
387  auto bearerInfo = m_bearerToInfoMap.find (bid)->second;
388 
389  LtePdcpSapProvider::TransmitPdcpSduParameters params;
390  params.pdcpSdu = packet;
391  params.rnti = bearerInfo->m_rnti;
392  params.lcid = lcid;
393 
394  NS_LOG_DEBUG(this << " MmWaveVehicularNetDevice::Send() bid " << (uint32_t)bid << " lcid " << (uint32_t)lcid << " rnti " << bearerInfo->m_rnti);
395 
396  packet->RemoveAllPacketTags (); // remove all tags in case there is any
397 
398  params.pdcpSdu = packet;
399  bearerInfo->m_pdcp->GetLtePdcpSapProvider()->TransmitPdcpSdu (params);
400 
401  return true;
402 }
