187 void
188 MmWaveSidelinkPhy::StartSlot (mmwave::SfnSf timingInfo)
189 {
190  NS_LOG_FUNCTION (this << " frame " << timingInfo.m_frameNum << " subframe " << timingInfo.m_sfNum << " slot " << timingInfo.m_slotNum);
191 
192  // trigger the MAC
193  m_phySapUser->SlotIndication (timingInfo);
194 
195  while (m_phyBuffer.size () != 0)
196  {
197  uint8_t usedSymbols = 0; // the symbol index
198 
199  // retrieve the first element in the list
200  Ptr<PacketBurst> pktBurst;
201  mmwave::TtiAllocInfo info;
202  std::tie (pktBurst, info) = m_phyBuffer.front ();
203 
204  // send the transport block
205  if (info.m_ttiType == mmwave::TtiAllocInfo::DATA)
206  {
207  usedSymbols += SlData (pktBurst, info);
208  }
209  else if (info.m_ttiType == mmwave::TtiAllocInfo::CTRL)
210  {
211  NS_FATAL_ERROR ("Control messages are not currently supported");
212  }
213  else
214  {
215  NS_FATAL_ERROR ("Unknown TB type");
216  }
217 
218  // check if we exceeded the slot boundaries
219  NS_ASSERT_MSG (usedSymbols <= m_phyMacConfig->GetSymbPerSlot (), "Exceeded number of available symbols");
220 
221  // remove the transport block from the buffer
222  m_phyBuffer.pop_front ();
223  }
224 
225  // update the timing information
226  timingInfo = UpdateTimingInfo (timingInfo);
227  Simulator::Schedule (m_phyMacConfig->GetSlotPeriod (), &MmWaveSidelinkPhy::StartSlot, this, timingInfo);
228 }

230 uint8_t
231 MmWaveSidelinkPhy::SlData (Ptr<PacketBurst> pb, mmwave::TtiAllocInfo info)
232 {
233  NS_LOG_FUNCTION (this);
234 
235  // create the tx PSD
236  //TODO do we need to create a new psd at each TTI?
237  std::vector<int> subChannelsForTx = SetSubChannelsForTransmission ();
238 
239  // compute the tx start time (IndexOfTheFirstSymbol * SymbolDuration)
240  Time startTime = info.m_dci.m_symStart * m_phyMacConfig->GetSymbolPeriod ();
241  NS_ASSERT_MSG (startTime == info.m_dci.m_symStart * m_phyMacConfig->GetSymbolPeriod (), "startTime was not been correctly set");
242 
243  // compute the duration of the transmission (NumberOfSymbols * SymbolDuration)
244  Time duration = info.m_dci.m_numSym * m_phyMacConfig->GetSymbolPeriod ();
245 
246  NS_ASSERT_MSG (duration == info.m_dci.m_numSym * m_phyMacConfig->GetSymbolPeriod (), "duration was not been correctly set");
247 
248  // send the transport block
249  Simulator::Schedule (startTime, &MmWaveSidelinkPhy::SendDataChannels, this,
250  pb,
251  duration,
252  info,
253  subChannelsForTx);
254 
255  return info.m_dci.m_numSym;
256 }

273 std::vector<int>
274 MmWaveSidelinkPhy::SetSubChannelsForTransmission ()
275  {
276  // create the transmission mask, use all the available subchannels
277  std::vector<int> subChannelsForTx (m_phyMacConfig->GetNumChunks ());
278  for (uint32_t i = 0; i < subChannelsForTx.size (); i++)
279  {
280  subChannelsForTx.at(i) = i;
281  }
282 
283  // create the tx PSD
284  Ptr<SpectrumValue> txPsd = mmwave::MmWaveSpectrumValueHelper::CreateTxPowerSpectralDensity (m_phyMacConfig, m_txPower, subChannelsForTx);
285 
286  // set the tx PSD in the spectrum phy
287  m_sidelinkSpectrumPhy->SetTxPowerSpectralDensity (txPsd);
288 
289  return subChannelsForTx;
290  }

292 mmwave::SfnSf
293 MmWaveSidelinkPhy::UpdateTimingInfo (mmwave::SfnSf info) const
294 {
295  NS_LOG_INFO (this);
296 
297  uint32_t nextSlot = info.m_slotNum + 1;
298  uint32_t nextSf = info.m_sfNum;
299  uint32_t nextFrame = info.m_frameNum;
300 
301  if (nextSlot == m_phyMacConfig->GetSlotsPerSubframe ())
302  {
303  nextSlot = 0;
304  ++nextSf;
305 
306  if (nextSf == m_phyMacConfig->GetSubframesPerFrame ())
307  {
308  nextSf = 0;
309  ++nextFrame;
310  }
311  }
312 
313  // update the mmwave::SfnSf structure
314  info.m_slotNum = nextSlot;
315  info.m_sfNum = nextSf;
316  info.m_frameNum = nextFrame;
317 
318  return info;
319 }

354 MmWaveSidelinkPhy::GenerateSinrReport (const SpectrumValue& sinr, uint16_t rnti, uint8_t numSym, uint32_t tbSize, uint8_t mcs)
355 {
356  NS_LOG_FUNCTION (this << rnti << (uint32_t)numSym << tbSize << (uint32_t)mcs);
