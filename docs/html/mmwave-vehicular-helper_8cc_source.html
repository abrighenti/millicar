104 void
105 MmWaveVehicularHelper::DoInitialize ()
106 {
107  NS_LOG_FUNCTION (this);
108 
109  // intialize the RNTI counter
110  m_rntiCounter = 0;
111 
112  // if the PHY layer configuration object was not set manually, create it
113  if (!m_phyMacConfig)
114  {
115  m_phyMacConfig = CreateObjectWithAttributes<mmwave::MmWavePhyMacCommon> ("Numerology", EnumValue (m_numerologyIndex),
116  "Bandwidth", DoubleValue (m_bandwidth));
117  }
118 
119  // create the channel
120  m_channel = CreateObject<SingleModelSpectrumChannel> ();
121  if (!m_propagationLossModelType.empty ())
122  {
123  ObjectFactory factory (m_propagationLossModelType);
124  m_channel->AddPropagationLossModel (factory.Create<PropagationLossModel> ());
125  }
126  if (!m_spectrumPropagationLossModelType.empty ())
127  {
128  ObjectFactory factory (m_spectrumPropagationLossModelType);
129  m_channel->AddSpectrumPropagationLossModel (factory.Create<SpectrumPropagationLossModel> ());
130  }
131  if (!m_propagationDelayModelType.empty ())
132  {
133  ObjectFactory factory (m_propagationDelayModelType);
134  m_channel->SetPropagationDelayModel (factory.Create<PropagationDelayModel> ());
135  }
136 
137  // 3GPP vehicular channel needs proper configuration
138  if (m_spectrumPropagationLossModelType == "ns3::MmWaveVehicularSpectrumPropagationLossModel")
139  {
140  Ptr<MmWaveVehicularSpectrumPropagationLossModel> threeGppSplm = DynamicCast<MmWaveVehicularSpectrumPropagationLossModel> (m_channel->GetSpectrumPropagationLossModel ());
141  PointerValue plm;
142  m_channel->GetAttribute ("PropagationLossModel", plm);
143 
144  Ptr<MmWaveVehicularPropagationLossModel> pathloss = DynamicCast<MmWaveVehicularPropagationLossModel> (plm.Get<PropagationLossModel> ());
145  pathloss->SetFrequency (m_phyMacConfig->GetCenterFrequency());
146 
147  threeGppSplm->SetPathlossModel (pathloss); // associate pathloss and fast fading models
148  threeGppSplm->SetFrequency (m_phyMacConfig->GetCenterFrequency()); // set correct value of frequency
149 
150  }
151 }

199 Ptr<MmWaveVehicularNetDevice>
200 MmWaveVehicularHelper::InstallSingleMmWaveVehicularNetDevice (Ptr<Node> node, uint16_t rnti)
201 {
202  NS_LOG_FUNCTION (this);
203 
204  // create the antenna
205  Ptr<MmWaveVehicularAntennaArrayModel> aam = CreateObject<MmWaveVehicularAntennaArrayModel> ();
206 
207  // create and configure the tx spectrum phy
208  Ptr<MmWaveSidelinkSpectrumPhy> ssp = CreateObject<MmWaveSidelinkSpectrumPhy> ();
209  NS_ASSERT_MSG (node->GetObject<MobilityModel> (), "Missing mobility model");
210  ssp->SetMobility (node->GetObject<MobilityModel> ());
211  ssp->SetAntenna (aam);
212  NS_ASSERT_MSG (m_channel, "First create the channel");
213  ssp->SetChannel (m_channel);
214 
215  // add the spectrum phy to the spectrum channel
216  m_channel->AddRx (ssp);
217 
218  // create and configure the chunk processor
219  Ptr<mmwave::mmWaveChunkProcessor> pData = Create<mmwave::mmWaveChunkProcessor> ();
220  pData->AddCallback (MakeCallback (&MmWaveSidelinkSpectrumPhy::UpdateSinrPerceived, ssp));
221  ssp->AddDataSinrChunkProcessor (pData);
222 
223  // create the phy
224  NS_ASSERT_MSG (m_phyMacConfig, "First set the configuration parameters");
225  Ptr<MmWaveSidelinkPhy> phy = CreateObject<MmWaveSidelinkPhy> (ssp, m_phyMacConfig);
226 
227  // connect the rx callback of the spectrum object to the sink
228  ssp->SetPhyRxDataEndOkCallback (MakeCallback (&MmWaveSidelinkPhy::Receive, phy));
229 
230  // connect the callback to report the SINR
231  ssp->SetSidelinkSinrReportCallback (MakeCallback (&MmWaveSidelinkPhy::GenerateSinrReport, phy));
232 
233  if(m_phyTraceHelper != 0)
234  {
235  ssp->SetSidelinkSinrReportCallback (MakeCallback (&MmWaveVehicularTracesHelper::McsSinrCallback, m_phyTraceHelper));
236  }
237 
238  // create the mac
239  Ptr<MmWaveSidelinkMac> mac = CreateObject<MmWaveSidelinkMac> (m_phyMacConfig);
240  mac->SetRnti (rnti);
241 
242  // connect phy and mac
243  phy->SetPhySapUser (mac->GetPhySapUser ());
244  mac->SetPhySapProvider (phy->GetPhySapProvider ());
245 
246  // create and configure the device
247  Ptr<MmWaveVehicularNetDevice> device = CreateObject<MmWaveVehicularNetDevice> (phy, mac);
248  node->AddDevice (device);
249  device->SetNode (node);
250  ssp->SetDevice (device);
251 
252  // connect the rx callback of the mac object to the rx method of the NetDevice
253  mac->SetForwardUpCallback(MakeCallback(&MmWaveVehicularNetDevice::Receive, device));
254 
255  // initialize the channel (if needed)
256  Ptr<MmWaveVehicularSpectrumPropagationLossModel> splm = DynamicCast<MmWaveVehicularSpectrumPropagationLossModel> (m_channel->GetSpectrumPropagationLossModel ());
257  if (splm)
258  splm->AddDevice (device, aam);
259 
260  return device;
261 }

263 void
264 MmWaveVehicularHelper::PairDevices (NetDeviceContainer devices)
265 {
266  NS_LOG_FUNCTION (this);
267 
268  // TODO update this part to enable a more flexible configuration of the
269  // scheduling pattern
270 
271  std::vector<uint16_t> pattern = CreateSchedulingPattern(devices);
272 
273  uint8_t bearerId = 1;
274 
275  for (NetDeviceContainer::Iterator i = devices.Begin (); i != devices.End (); ++i)
276  {
277 
278  Ptr<MmWaveVehicularNetDevice> di = DynamicCast<MmWaveVehicularNetDevice> (*i);
279  Ptr<Node> iNode = di->GetNode ();
280  Ptr<Ipv4> iNodeIpv4 = iNode->GetObject<Ipv4> ();
281  NS_ASSERT_MSG (iNodeIpv4 != 0, "Nodes need to have IPv4 installed before pairing can be activated");
282 
283  di->GetMac ()->SetSfAllocationInfo (pattern); // this is called ONCE for each NetDevice
284 
285  for (NetDeviceContainer::Iterator j = i + 1; j != devices.End (); ++j)
286  {
287  Ptr<MmWaveVehicularNetDevice> dj = DynamicCast<MmWaveVehicularNetDevice> (*j);
288  Ptr<Node> jNode = dj->GetNode ();
289  Ptr<Ipv4> jNodeIpv4 = jNode->GetObject<Ipv4> ();
290  NS_ASSERT_MSG (jNodeIpv4 != 0, "Nodes need to have IPv4 installed before pairing can be activated");
291 
292  // initialize the <IP address, RNTI> map of the devices
293  int32_t interface = jNodeIpv4->GetInterfaceForDevice (dj);
294  Ipv4Address diAddr = iNodeIpv4->GetAddress (interface, 0).GetLocal ();
295  Ipv4Address djAddr = jNodeIpv4->GetAddress (interface, 0).GetLocal ();
296 
297  // register the associated devices in the PHY
298  di->GetPhy ()->AddDevice (dj->GetMac ()->GetRnti (), dj);
299  dj->GetPhy ()->AddDevice (di->GetMac ()->GetRnti (), di);
300 
301  // bearer activation by creating a logical channel between the two devices
302  NS_LOG_DEBUG("Activation of bearer between " << diAddr << " and " << djAddr);
303  NS_LOG_DEBUG("Bearer ID: " << uint32_t(bearerId) << " - Associate RNTI " << di->GetMac ()->GetRnti () << " to " << dj->GetMac ()->GetRnti ());
304 
305  di->ActivateBearer(bearerId, dj->GetMac ()->GetRnti (), djAddr);
306  dj->ActivateBearer(bearerId, di->GetMac ()->GetRnti (), diAddr);
307  bearerId++;
308  }
309 
310 
311  }

314 std::vector<uint16_t>
315 MmWaveVehicularHelper::CreateSchedulingPattern (NetDeviceContainer devices)
316 {
317  // The maximum supported number of vehicles in the group is equal to the available number of slots per subframe
318  // TODO implement scheduling pattern that support groups with a number of vehicle greater than the number of slots per subframe
319  NS_ABORT_MSG_IF (devices.GetN () > m_phyMacConfig->GetSlotsPerSubframe (), "Too many devices");
320 
321  // NOTE fixed scheduling pattern, set in configuration time and assumed the same for each subframe
322 
323  uint8_t slotPerSf = m_phyMacConfig->GetSlotsPerSubframe ();
324  std::vector<uint16_t> pattern;
325 
326  switch (m_schedulingOpt)
327  {
328  case DEFAULT:
329  {
330  // Each slot in the subframe is assigned to a different user.
331  // If (numDevices < numSlots), the remaining available slots are unused
332  pattern = std::vector<uint16_t> (slotPerSf);
333  for (uint16_t i = 0; i < devices.GetN (); i++)
334  {
335  Ptr<MmWaveVehicularNetDevice> di = DynamicCast<MmWaveVehicularNetDevice> (devices.Get (i));
336  pattern.at(i) = di->GetMac ()->GetRnti ();
337  NS_LOG_DEBUG ("slot " << i << " assigned to rnti " << di->GetMac ()->GetRnti ());
338  }
339  break;
340  }
341  case OPTIMIZED:
342  {
343  // Each slot in the subframe is used
344  uint8_t slotPerDev = std::floor ( slotPerSf / devices.GetN ());
345  uint8_t remainingSlots = slotPerSf % devices.GetN ();
346 
347  NS_LOG_DEBUG("Minimum number of slots per device = " << (uint16_t)slotPerDev);
348  NS_LOG_DEBUG("Available slots = " << (uint16_t)slotPerSf);
349 
350  uint8_t slotCnt = 0;
351 
352  for (uint16_t i = 0; i < devices.GetN (); i++)
353  {
354  Ptr<MmWaveVehicularNetDevice> di = DynamicCast<MmWaveVehicularNetDevice> (devices.Get (i));
355 
356  for (uint8_t j = 0; j < slotPerDev; j++)
357  {
358  pattern.push_back(di->GetMac ()->GetRnti ());
359  NS_LOG_DEBUG ("slot " << uint16_t(slotCnt) << " assigned to rnti " << di->GetMac ()->GetRnti ());
360  slotCnt++;
361  }
362  }
363 
364  NS_LOG_DEBUG("Remaining slots = " << (uint16_t)remainingSlots);
365  for (uint16_t i = 0; i < remainingSlots; i++)
366  {
367  Ptr<MmWaveVehicularNetDevice> di = DynamicCast<MmWaveVehicularNetDevice> (devices.Get (i));
368  pattern.push_back(di->GetMac ()->GetRnti ());
369  NS_LOG_DEBUG ("slot " << uint16_t(slotCnt) << " assigned to rnti " << di->GetMac ()->GetRnti ());
370  slotCnt++;
371  }
372  break;
373  }
374  default:
375  {
376  NS_FATAL_ERROR("Programming Error.");
377  }
378  }
379 
380  return pattern;
381 }
