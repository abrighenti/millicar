131 void
132 MmWaveVehicularInterferenceTestCase::StartTest (double txPower)
133 {
134  uint32_t packetSize = 1024; // bytes
135  Time startTime = Seconds (1.5);
136  Time endTime = Seconds (4.0);
137  uint8_t mcs = 12;
138  double speed = 20; // m/s
139 
140  Config::SetDefault ("ns3::MmWaveSidelinkMac::Mcs", UintegerValue (mcs));
141  Config::SetDefault ("ns3::MmWavePhyMacCommon::CenterFreq", DoubleValue (60.0e9));
142 
143  // create the nodes
144  NodeContainer group1, group2;
145  group1.Create (2);
146  group2.Create (2);
147 
148  // create the mobility models
149  MobilityHelper mobility;
150  mobility.SetMobilityModel ("ns3::ConstantVelocityMobilityModel");
151  mobility.Install (group1);
152  mobility.Install (group2);
153 
154  group1.Get (0)->GetObject<MobilityModel> ()->SetPosition (Vector (0,0,0));
155  group1.Get (0)->GetObject<ConstantVelocityMobilityModel> ()->SetVelocity (Vector (0, speed, 0));
156 
157  group1.Get (1)->GetObject<MobilityModel> ()->SetPosition (Vector (0,20,0));
158  group1.Get (1)->GetObject<ConstantVelocityMobilityModel> ()->SetVelocity (Vector (0, speed, 0));
159 
160  group2.Get (0)->GetObject<MobilityModel> ()->SetPosition (Vector (20,120,0));
161  group2.Get (0)->GetObject<ConstantVelocityMobilityModel> ()->SetVelocity (Vector (0, -speed, 0));
162 
163  group2.Get (1)->GetObject<MobilityModel> ()->SetPosition (Vector (20,140,0));
164  group2.Get (1)->GetObject<ConstantVelocityMobilityModel> ()->SetVelocity (Vector (0, -speed, 0));
165 
166  // create and configure the helper
167  Ptr<MmWaveVehicularHelper> helper = CreateObject<MmWaveVehicularHelper> ();
168  helper->SetNumerology (3);
169  NetDeviceContainer devs1 = helper->InstallMmWaveVehicularNetDevices (group1);
170  NetDeviceContainer devs2 = helper->InstallMmWaveVehicularNetDevices (group2);
171 
172  // Only the tx power of vehicles of the second group is tuned.
173  DynamicCast<MmWaveVehicularNetDevice>(devs2.Get(0))->GetPhy()->SetTxPower(txPower);
174 
175  Ptr<MmWaveSidelinkPhy> slPhy_2 = DynamicCast<MmWaveVehicularNetDevice>(devs2.Get(1))->GetPhy();
176  slPhy_2->SetTxPower(txPower);
177 
178  InternetStackHelper internet;
179  internet.Install (group1);
180  internet.Install (group2);
181 
182  Ipv4AddressHelper ipv4;
183  NS_LOG_INFO ("Assign IP Addresses.");
184 
185  ipv4.SetBase ("10.1.1.0", "255.255.255.0");
186  Ipv4InterfaceContainer i = ipv4.Assign (devs1);
187 
188  ipv4.SetBase ("10.1.2.0", "255.255.255.0");
189  i = ipv4.Assign (devs2);
190 
191  helper->PairDevices(devs1);
192  helper->PairDevices(devs2);
193 
194  Ipv4StaticRoutingHelper ipv4RoutingHelper;
195 
196  Ptr<Ipv4StaticRouting> staticRouting = ipv4RoutingHelper.GetStaticRouting (group1.Get (0)->GetObject<Ipv4> ());
197  staticRouting->SetDefaultRoute (group1.Get (1)->GetObject<Ipv4> ()->GetAddress (1, 0).GetLocal () , 2 );
198 
199  NS_LOG_DEBUG("IPv4 Address node 0 group 1: " << group1.Get (0)->GetObject<Ipv4> ()->GetAddress (1, 0).GetLocal ());
200  NS_LOG_DEBUG("IPv4 Address node 1 group 1: " << group1.Get (1)->GetObject<Ipv4> ()->GetAddress (1, 0).GetLocal ());
201 
202  staticRouting = ipv4RoutingHelper.GetStaticRouting (group2.Get (0)->GetObject<Ipv4> ());
203  staticRouting->SetDefaultRoute (group2.Get (1)->GetObject<Ipv4> ()->GetAddress (1, 0).GetLocal () , 2 );
204 
205  NS_LOG_DEBUG("IPv4 Address node 0 group 2: " << group2.Get (0)->GetObject<Ipv4> ()->GetAddress (1, 0).GetLocal ());
206  NS_LOG_DEBUG("IPv4 Address node 1 group 2: " << group2.Get (1)->GetObject<Ipv4> ()->GetAddress (1, 0).GetLocal ());
207 
208  Ptr<MmWaveAmc> m_amc = CreateObject <MmWaveAmc> (helper->GetConfigurationParameters());
209  double availableRate = m_amc->GetTbSizeFromMcsSymbols(mcs, 14) / 0.001; // bps
210  uint16_t port = 4000; // well-known echo port number
211  uint32_t maxPacketCount = 1;
212  packetSize = m_amc->GetTbSizeFromMcsSymbols(mcs, 14) / 8 - 28;
213  Time interPacketInterval = Seconds(double((packetSize * 8) / availableRate));
214 
215  NS_LOG_INFO ("Create applications for group number 1.");
216 
217  UdpServerHelper server1 (port);
218 
219  ApplicationContainer apps1 = server1.Install (group1.Get (1));
220  apps1.Start (Seconds (1.0));
221  apps1.Stop (Seconds(4.0));
222 
223  UdpEchoClientHelper client1 (group1.Get (1)->GetObject<Ipv4> ()->GetAddress (1, 0).GetLocal (), port);
224  client1.SetAttribute ("MaxPackets", UintegerValue (maxPacketCount));
225  client1.SetAttribute ("Interval", TimeValue (interPacketInterval));
226  client1.SetAttribute ("PacketSize", UintegerValue (packetSize));
227  apps1 = client1.Install (group1.Get (0));
228  apps1.Start (startTime);
229  apps1.Stop (endTime);
230 
231  NS_LOG_INFO ("Create applications for group number 2.");
232 
233  UdpServerHelper server2 (port);
234 
235  ApplicationContainer apps2 = server2.Install (group2.Get (1));
236  apps2.Start (Seconds (1.0));
237  apps2.Stop (Seconds(4.0));
238 
239  UdpEchoClientHelper client2 (group2.Get (1)->GetObject<Ipv4> ()->GetAddress (1, 0).GetLocal (), port);
240  client2.SetAttribute ("MaxPackets", UintegerValue (maxPacketCount));
241  client2.SetAttribute ("Interval", TimeValue (interPacketInterval));
242  client2.SetAttribute ("PacketSize", UintegerValue (packetSize));
243  apps2 = client2.Install (group2.Get (0));
244  apps2.Start (startTime);
245  apps2.Stop (endTime);
246 
247  // ---------------------------------
248 
249  double kT_dBm_Hz = -174.0; // dBm/Hz
250  double kT_W_Hz = std::pow (10.0, (kT_dBm_Hz - 30) / 10.0);
251  double noiseFigureLinear = std::pow (10.0, 5 / 10.0); // Noise Figure is equal to 5 dB by default
252  double noisePowerSpectralDensity = kT_W_Hz * noiseFigureLinear;
253  double noisePower = noisePowerSpectralDensity * 1e8; // 1e8 is the bandwidth used for transmission
254 
255  // Transmitted Power is 30 dBm by default, which is the equivalent to 1 W
256  double txPowerLinear1 = std::pow (10.0, (30 - 30) / 10.0);
257  double txPowerLinear2 = std::pow (10.0, (txPower - 30) / 10.0);
258  std::cout << "---- GROUP 1 ----" << std::endl;
259  std::cout << "Transmitted Power: " << txPowerLinear1 << "W" << std::endl;
260  std::cout << "Noise Power: "<< noisePower << std::endl;
261  g_expectedSinrG1 = txPowerLinear1/(txPowerLinear2+noisePower);
262  std::cout << "Expected SINR: "<< g_expectedSinrG1 << std::endl;
263 
264  // create and configure the chunk processor
265  Ptr<MmWaveSidelinkPhy> slPhy_1 = DynamicCast<MmWaveVehicularNetDevice>(devs1.Get(1))->GetPhy();
266  Ptr<mmWaveChunkProcessor> pData_1 = Create<mmWaveChunkProcessor> ();
267  pData_1->AddCallback (MakeCallback (&MmWaveVehicularInterferenceTestCase::CheckSinrPerceivedG1, this));
268  slPhy_1->GetSpectrumPhy()->AddDataSinrChunkProcessor (pData_1);
269 
270  std::cout << "---- GROUP 2 ----" << std::endl;
271  std::cout << "Transmitted Power: " << txPowerLinear2 << "W" << std::endl;
272  std::cout << "Noise Power: "<< noisePower << std::endl;
273  g_expectedSinrG2 = txPowerLinear2/(txPowerLinear1+noisePower);
274  std::cout << "Expected SINR: "<< g_expectedSinrG2 << std::endl;
275 
276  Ptr<mmWaveChunkProcessor> pData_2 = Create<mmWaveChunkProcessor> ();
277  pData_2->AddCallback (MakeCallback (&MmWaveVehicularInterferenceTestCase::CheckSinrPerceivedG2, this));
278  slPhy_2->GetSpectrumPhy()->AddDataSinrChunkProcessor (pData_2);
279 
280  Simulator::Stop (Seconds(4.0));
281  Simulator::Run ();
282  Simulator::Destroy ();
283
