137 double
138 MmWaveVehicularPropagationLossModel::GetLoss (Ptr<MobilityModel> deviceA, Ptr<MobilityModel> deviceB) const
139 {
140  NS_ASSERT_MSG (m_frequency != 0.0, "Set the operating frequency first!");
141 
142  Vector aPos = deviceA->GetPosition ();
143  Vector bPos = deviceB->GetPosition ();
144  double x = aPos.x - bPos.x;
145  double y = aPos.y - bPos.y;
146  double distance2D = sqrt (x * x + y * y);
147  double hA = aPos.z;
148  double hB = bPos.z;
149 
150  double distance3D = deviceA->GetDistanceFrom (deviceB);
151 
152  if (distance3D < 3 * m_lambda)
153  {
154  NS_LOG_UNCOND ("distance not within the far field region => inaccurate propagation loss value");
155  }
156  if (distance3D <= 0)
157  {
158  return m_minLoss;
159  }
160 
161  channelConditionMap_t::const_iterator it;
162  it = m_channelConditionMap.find (std::make_pair (deviceA, deviceB));
163  if (it == m_channelConditionMap.end ())
164  {
165  channelCondition condition;
166 
167  if (m_channelConditions.compare ("l") == 0 )
168  {
169  condition.m_channelCondition = 'l';
170  NS_LOG_UNCOND (m_scenario << " scenario, channel condition is fixed to be " << condition.m_channelCondition << ", h_A=" << hA << ",h_B=" << hB);
171  }
172  else if (m_channelConditions.compare ("n") == 0)
173  {
174  condition.m_channelCondition = 'n';
175  NS_LOG_UNCOND (m_scenario << " scenario, channel condition is fixed to be " << condition.m_channelCondition << ", h_A=" << hA << ",h_B=" << hB);
176  }
177  else if (m_channelConditions.compare ("v") == 0)
178  {
179  condition.m_channelCondition = 'v';
180  NS_LOG_UNCOND (m_scenario << " scenario, channel condition is fixed to be " << condition.m_channelCondition << ", h_A=" << hA << ",h_B=" << hB);
181  }
182  else if (m_channelConditions.compare ("a") == 0)
183  {
184  double PRef = m_uniformVar->GetValue ();
185  double probLos, probnLos, probnLosv;
186 
187  if (m_scenario == "V2V-Highway")
188  {
189  double a, b, c;
190  a = 2.1013e-6;
191  b = - 0.002;
192  c = 1.0193;
193 
194  if (distance3D <= 475)
195  {
196  probLos = std::min(1.0, a * pow(distance3D, 2) + b * distance3D + c);
197  }
198  else
199  {
200  probLos = std::max(0.0, 0.54 - 0.001 * (distance3D - 475));
201  }
202 
203  if (PRef <= probLos)
204  {
205  condition.m_channelCondition = 'l';
206  }
207  else
208  {
209  condition.m_channelCondition = 'v';
210  }
211  }
212  else if (m_scenario == "V2V-Urban")
213  {
214  probLos = std::min(1.0, 1.05 * exp(-0.0114 * distance3D));
215 
216  if (PRef <= probLos)
217  {
218  condition.m_channelCondition = 'l';
219  }
220  else
221  {
222  condition.m_channelCondition = 'v';
223  }
224  }
225  else if (m_scenario == "Extended-V2V-Highway")
226  {
227  // As established from TR 37.885 we have to define
228  double aLOS, bLOS, cLOS = 1;
229  aLOS = 2.7e-6;
230  bLOS = - 0.0025;
231 
232  probLos = std::min(1.0, std::max(0.0, aLOS * pow(distance3D, 2) + bLOS * distance3D + cLOS));
233 
234  double aNLOS, bNLOS, cNLOS = 0.015;
235  aNLOS = -3.7e-7;
236  bNLOS = 0.00061;
237 
238  probnLos = std::min(1.0, std::max(0.0, aNLOS * pow(distance3D, 2) + bNLOS * distance3D + cNLOS));
239 
240  if (PRef <= probLos)
241  {
242  condition.m_channelCondition = 'l';
243  }
244  else if (PRef <= probLos + probnLos)
245  {
246  condition.m_channelCondition = 'n';
247  }
248  else
249  {
250  condition.m_channelCondition = 'v';
251  }
252  }
253  else if (m_scenario == "Extended-V2V-Urban")
254  {
255  probLos = std::min(1.0, std::max(0.0, 0.8372 * exp (-0.0114*distance3D)));
256  probnLosv = std::min(1.0, std::max(0.0, 1/(0.0312*distance3D) * exp(- pow(log(distance3D) - 5.0063, 2) / 2.4544)));
257 
258  if (PRef <= probLos)
259  {
260  condition.m_channelCondition = 'l';
261  }
262  else if (PRef <= probLos + probnLosv)
263  {
264  condition.m_channelCondition = 'v';
265  }
266  else
267  {
268  condition.m_channelCondition = 'n';
269  }
270  }
271  else
272  {
273  NS_FATAL_ERROR ("Unknown scenario");
274  }
275 
276  NS_LOG_DEBUG (m_scenario << " scenario, 2D distance = " << distance2D << "m, Prob_LOS = " << probLos
277  << ", Prob_REF = " << PRef << ", the channel condition is " << condition.m_channelCondition << ", h_A=" << hA << ",h_B=" << hB);
278  }
279  else
280  {
281  NS_FATAL_ERROR ("Wrong channel condition configuration");
282  }
283 
284  // assign a large negative value to identify initial transmission.
285  condition.m_shadowing = -1e6;
286 
287  std::pair<channelConditionMap_t::const_iterator, bool> ret;
288  ret = m_channelConditionMap.insert (std::make_pair (std::make_pair (deviceA, deviceB), condition));
289  m_channelConditionMap.insert (std::make_pair (std::make_pair (deviceB, deviceA), condition));
290  it = ret.first;
291  }
292 
293  double lossDb = 0;
294  double freqGHz = m_frequency / 1e9;
295 
296  double shadowingStd = 0;
297  double shadowingCorDistance = 0;
298 
299  if (m_scenario == "V2V-Highway")
300  {
301  // The shadowing standard deviation and decorrelation distance are
302  // specified in TR 36.885 Sec. A.1.4
303  shadowingStd = 3.0;
304  shadowingCorDistance = 25.0;
305 
306  switch ((*it).second.m_channelCondition)
307  {
308  case 'l':
309  {
310  lossDb = 32.4 + 20 * log10 (distance3D) + 20 * log10 (freqGHz);
311  break;
312  }
313  case 'v':
314  {
315  double additionalLoss = GetAdditionalNlosVLoss (distance3D, hA, hB);
316  lossDb = 32.4 + 20 * log10 (distance3D) + 20 * log10 (freqGHz) + additionalLoss;
317  break;
318  }
319  case 'n':
320  {
321  lossDb = 36.85 + 30 * log10 (distance3D) + 18.9 * log10 (freqGHz);
322  break;
323  }
324  default:
325  NS_FATAL_ERROR ("Programming Error.");
326 
327  }
328  }
329  else if (m_scenario == "V2V-Urban")
330  {
331  switch ((*it).second.m_channelCondition)
332  {
333  case 'l':
334  {
335  lossDb = 38.77 + 16.7 * log10 (distance3D) + 18.2 * log10 (freqGHz);
336 
337  // The shadowing standard deviation and decorrelation distance are
338  // specified in TR 36.885 Sec. A.1.4
339  shadowingStd = 3.0;
340  shadowingCorDistance = 10.0;
341  break;
342  }
343  case 'v':
344  {
345  double additionalLoss = GetAdditionalNlosVLoss (distance3D, hA, hB);
346 
347  lossDb = 38.77 + 16.7 * log10 (distance3D) + 18.2 * log10 (freqGHz) + additionalLoss;
348  break;
349 
350  }
351  case 'n':
352  {
353  lossDb = 36.85 + 30 * log10 (distance3D) + 18.9 * log10 (freqGHz);
354 
355  // The shadowing standard deviation and decorrelation distance are
356  // specified in TR 36.885 Sec. A.1.4
357  shadowingStd = 4.0;
358  shadowingCorDistance = 10.0;
359  break;
360  }
361  default:
362  NS_FATAL_ERROR ("Programming Error.");
363 
364  }
365  }
366  else if (m_scenario == "Extended-V2V-Highway")
367  {
368  switch ((*it).second.m_channelCondition)
369  {
370  case 'l':
371  {
372  lossDb = 32.4 + 20 * log10 (distance3D) + 20 * log10 (freqGHz);
373 
374  shadowingStd = 3.0;
375  // The extended model does not specify the decorrelation distance. I
376  // assume the one specified by TR 36.885
377  shadowingCorDistance = 25.0;
378  break;
379  }
380  case 'v':
381  {
382  double additionalLoss = GetAdditionalNlosVLoss (distance3D, hA, hB);
383 
384  lossDb = 32.4 + 20 * log10 (distance3D) + 20 * log10 (freqGHz) + additionalLoss;
385  break;
386  }
387  case 'n':
388  {
389  lossDb = 36.85 + 30 * log10 (distance3D) + 18.9 * log10 (freqGHz);
390  shadowingStd = 4.0;
391  // The extended model does not specify the decorrelation distance. I
392  // assume the one specified by TR 36.885
393  shadowingCorDistance = 25.0;
394  break;
395  }
396  default:
397  NS_FATAL_ERROR ("Programming Error.");
398  }
399  }
400  else if (m_scenario == "Extended-V2V-Urban")
401  {
402  switch ((*it).second.m_channelCondition)
403  {
404  case 'l':
405  {
406  lossDb = 38.77 + 16.7 * log10 (distance3D) + 18.2 * log10 (freqGHz);
407  shadowingStd = 3.0;
408  // The extended model does not specify the decorrelation distance. I
409  // assume the one specified by TR 36.885
410  shadowingCorDistance = 10.0;
411  break;
412  }
413  case 'v':
414  {
415  double additionalLoss = GetAdditionalNlosVLoss (distance3D, hA, hB);
416 
417  lossDb = 38.77 + 16.7 * log10 (distance3D) + 18.2 * log10 (freqGHz) + additionalLoss;
418  break;
419  }
420  case 'n':
421  {
422  lossDb = 36.85 + 30 * log10 (distance3D) + 18.9 * log10 (freqGHz);
423  shadowingStd = 4.0;
424  // The extended model does not specify the decorrelation distance. I
425  // assume the one specified by TR 36.885
426  shadowingCorDistance = 10.0;
427  break;
428  }
429  default:
430  NS_FATAL_ERROR ("Programming Error.");
431  }
432  }
433  else
434  {
435  NS_FATAL_ERROR ("Unknown channel scenario");
436  }
437 
438  if (m_shadowingEnabled)
439  {
440 
441  channelCondition cond;
442  cond = (*it).second;
443 
444  //The first transmission the shadowing is initialized as -1e6,
445  //we perform this if check to identify the first transmission.
446  m_logNorVar->SetAttribute ("Sigma", DoubleValue (shadowingStd));
447 
448  if ((*it).second.m_shadowing < -1e5)
449  {
450  cond.m_shadowing = m_norVar->GetValue () * shadowingStd;
451  }
452  else
453  {
454  double deltaX = aPos.x - (*it).second.m_position.x;
455  double deltaY = aPos.y - (*it).second.m_position.y;
456  double disDiff = sqrt (deltaX * deltaX + deltaY * deltaY);
457  double R = exp (-1 * disDiff / shadowingCorDistance);
458 
459  cond.m_shadowing = R * (*it).second.m_shadowing + sqrt (1 - R * R) * m_norVar->GetValue () * shadowingStd;
460  }
461 
462  lossDb += cond.m_shadowing;
463  cond.m_position = deviceA->GetPosition ();
464  UpdateConditionMap (deviceA,deviceB,cond);
465  }
466 
467  return std::max (lossDb, m_minLoss);
468 }

470 double
471 MmWaveVehicularPropagationLossModel::GetAdditionalNlosVLoss (double distance3D, double hA, double hB) const
472 {
473  // From TR 37.885 v15.2.0
474  // When a V2V link is in NLOSv, additional vehicle blockage loss is
475  // added as follows:
476  // 1. The blocker height is the vehicle height which is randomly selected
477  // out of the three vehicle types according to the portion of the vehicle
478  // types in the simulated scenario.
479  double additionalLoss = 0;
480  double blockerHeight = 0;
481  double mu_a = 0;
482  double sigma_a = 0;
483  double randomValue = m_uniformVar->GetValue () * 3.0;
484  if (randomValue < m_percType3Vehicles)
485  {
486  // vehicles of type 3 have height 3 meters
487  blockerHeight = 3.0;
488  }
489  else
490  {
491  // vehicles of type 1 and 2 have height 1.6 meters
492  blockerHeight = 1.6;
493  }
494 
495  // The additional blockage loss is max {0 dB, a log-normal random variable}
496  if (std::min (hA, hB) > blockerHeight)
497  {
498  // Case 1: Minimum antenna height value of TX and RX > Blocker height
499  additionalLoss = 0;
500  }
501  else if (std::max (hA, hB) < blockerHeight)
502  {
503  // Case 2: Maximum antenna height value of TX and RX < Blocker height
504  mu_a = 9.0 + std::max (0.0, 15 * log10 (distance3D) - 41.0);
505  sigma_a = 4.5;
506  // Pay attention to the ambiguous definition of the parameters.
507  // Vehicular TR 37.885 defines mu_a and sigma_a as the mean and standard deviation of the log-normal random variable.
508  // ns-3's RNG considers mu and sigma as specific parameters of the log-normal distribution, while the mean and standard deviation are evaluated separately.
509  m_logNorVar->SetAttribute ("Mu", DoubleValue (log10(pow(mu_a,2) / sqrt(pow(sigma_a,2) + pow(mu_a,2)))));
510  m_logNorVar->SetAttribute ("Sigma", DoubleValue (sqrt(log10(pow(sigma_a,2) / pow(mu_a,2) + 1))));
511  additionalLoss = std::max(0.0, m_logNorVar->GetValue());
512  }
513  else
514  {
515  // Case 3: Otherwise
516  mu_a = 5.0 + std::max (0.0, 15 * log10 (distance3D) - 41.0);
517  sigma_a = 4.0;
518  // Pay attention to the ambiguous definition of the parameters.
519  // Vehicular TR 37.885 defines mu_a and sigma_a as the mean and standard deviation of the log-normal random variable.
520  // ns-3's RNG considers mu and sigma as specific parameters of the log-normal distribution, while the mean and standard deviation are evaluated separately.
521  m_logNorVar->SetAttribute ("Mu", DoubleValue (log10(pow(mu_a,2) / sqrt(pow(sigma_a,2) + pow(mu_a,2)))));
522  m_logNorVar->SetAttribute ("Sigma", DoubleValue (sqrt(log10(pow(sigma_a,2) / pow(mu_a,2) + 1))));
523  additionalLoss = std::max(0.0, m_logNorVar->GetValue());
524  }
525 
526  return additionalLoss;
527 }
