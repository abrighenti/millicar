258 void
259 MmWaveSidelinkSpectrumPhy::StartRxData (Ptr<MmWaveSidelinkSpectrumSignalParameters> params)
260 {
261  NS_LOG_FUNCTION (this);
262 
263  switch (m_state)
264  {
265  case TX:
266  // If there are other intereferent devices that transmit in the same slot, the current
267  // device simply does not consider the signal and goes on with the transmission. The code does not raise any errors since
268  // otherwise we are not able to study scenarios where interference could be an issue.
269  break;
270  case RX_CTRL:
271  NS_FATAL_ERROR ("Cannot receive control in data period");
272  break;
273  case RX_DATA:
274  // If this device is in the RX_DATA state and another call to StartRx is
275  // triggered, it means that multiple concurrent signals are being received.
276  // In this case, we assume that the device will synchronize with the first
277  // received signal, while the other will act as interferers
278  m_interferenceData->AddSignal (params->psd, params->duration);
279  break;
280  case IDLE:
281  {
282  // check if the packet is for this device, otherwise
283  // consider it only for the interference
284  m_interferenceData->AddSignal (params->psd, params->duration);
285  uint16_t thisDeviceRnti =
286  DynamicCast<MmWaveVehicularNetDevice>(m_device)->GetMac()->GetRnti();
287  if(thisDeviceRnti == params->destinationRnti)
288  {
289  // this is a useful signal
290  m_interferenceData->StartRx (params->psd);
291 
292  if (m_rxTransportBlock.empty ())
293  {
294  NS_ASSERT (m_state == IDLE);
295  // first transmission, i.e., we're IDLE and we start RX
296  m_firstRxStart = Simulator::Now ();
297  m_firstRxDuration = params->duration;
298  NS_LOG_LOGIC (this << " scheduling EndRx with delay " << params->duration.GetSeconds () << "s");
299 
300  m_endRxDataEvent = Simulator::Schedule (params->duration, &MmWaveSidelinkSpectrumPhy::EndRxData, this);
301  }
302  else
303  {
304  NS_ASSERT (m_state == RX_DATA);
305  // sanity check: if there are multiple RX events, they
306  // should occur at the same time and have the same
307  // duration, otherwise the interference calculation
308  // won't be correct
309  NS_ASSERT ((m_firstRxStart == Simulator::Now ()) && (m_firstRxDuration == params->duration));
310  }
311 
312  ChangeState (RX_DATA);
313  if (params->packetBurst && !params->packetBurst->GetPackets ().empty ())
314  {
315  TbInfo_t tbInfo = {params->packetBurst, params->size, params->mcs, params->numSym, params->senderRnti, params->rbBitmap};
316  m_rxTransportBlock.push_back (tbInfo);
317  }
318  }
319  else
320  {
321  NS_LOG_LOGIC (this << " not in sync with this signal (rnti="
322  << params->destinationRnti << ", rnti of the device="
323  << thisDeviceRnti << ")");
324  }
325  //m_rxControlMessageList.insert (m_rxControlMessageList.end (), params->ctrlMsgList.begin (), params->ctrlMsgList.end ());
326  }
327  break;
328  default:
329  NS_FATAL_ERROR ("Programming Error: Unknown State");
330  }
331 }

370 void
371 MmWaveSidelinkSpectrumPhy::EndRxData ()
372 {
373  NS_LOG_FUNCTION (this);
374  m_interferenceData->EndRx ();
375 
376  double sinrAvg = Sum (m_sinrPerceived) / (m_sinrPerceived.GetSpectrumModel ()->GetNumBands ());
377 
378  NS_ASSERT (m_state = RX_DATA);
379 
380  if ((m_dataErrorModelEnabled)&&(m_rxTransportBlock.size () > 0))
381  {
382  for (std::list<TbInfo_t>::const_iterator i = m_rxTransportBlock.begin ();
383  i != m_rxTransportBlock.end (); ++i)
384  {
385  // Here we need to initialize an empty harqInfoList since it is mandatory input
386  // for the method. Since the vector is empty, no harq procedures are triggeres (as we want)
387  std::vector <mmwave::MmWaveHarqProcessInfoElement_t> harqInfoList;
388 
389  NS_LOG_DEBUG ("average sinr " << 10*log10 (sinrAvg) << " MCS " << (uint16_t)(*i).mcs);
390  mmwave::MmWaveTbStats_t tbStats = mmwave::MmWaveMiErrorModel::GetTbDecodificationStats (m_sinrPerceived, (*i).rbBitmap, (*i).size, (*i).mcs, harqInfoList);
391 
392  // trigger callbacks
393  for (auto& it : m_slSinrReportCallback)
394  {
395  it(m_sinrPerceived, (*i).rnti, (*i).numSym, (*i).size, (*i).mcs); // TODO also export corrupt and tbler)
396  }
397 
398  bool corrupt = m_random->GetValue () > tbStats.tbler ? false : true;
399  if(!corrupt)
400  {
401  Ptr<PacketBurst> burst = (*i).packetBurst;
402  for (std::list<Ptr<Packet> >::const_iterator j = (*burst).Begin (); j != (*burst).End (); ++j)
403  {
404  if ((*j)->GetSize () == 0)
405  {
406  continue;
407  }
408 
409  // Do we need the LteRadioBearerTag also here to check the rnti? I don't think so.
410  NS_ASSERT_MSG (!m_phyRxDataEndOkCallback.IsNull (), "First set the rx callback");
411  m_phyRxDataEndOkCallback (*j);
412  }
413  }
414  else
415  {
416  NS_LOG_INFO ("TB failed");
417  }
418  }
419  }
420 
421  // forward control messages of this frame to MmWavePhy
422 
423  // if (!m_rxControlMessageList.empty () && !m_phyRxCtrlEndOkCallback.IsNull ())
424  // {
425  // m_phyRxCtrlEndOkCallback (m_rxControlMessageList);
426  // }
427 
428  m_state = IDLE;
429  m_rxTransportBlock.clear ();
430  //m_rxControlMessageList.clear ();
431 }

452 bool
453 MmWaveSidelinkSpectrumPhy::StartTxDataFrames (Ptr<PacketBurst> pb,
454  Time duration,
455  uint8_t mcs,
456  uint32_t size,
457  uint8_t numSym,
458  uint16_t senderRnti,
459  uint16_t destinationRnti,
460  std::vector<int> rbBitmap)
461 {
462  NS_LOG_FUNCTION (this);
463 
464  NS_ASSERT_MSG (m_channel, "First configure the SpectrumChannel");
465 
466  switch (m_state)
467  {
468  case RX_DATA:
469  case RX_CTRL:
470  NS_FATAL_ERROR ("Cannot transmit while receiving");
471  break;
472  case TX:
473  NS_FATAL_ERROR ("Cannot transmit while a different transmission is still on");
474  break;
475  case IDLE:
476  {
477  NS_ASSERT (m_txPsd);
478 
479  m_state = TX;
480  Ptr<MmWaveSidelinkSpectrumSignalParameters> txParams = Create<MmWaveSidelinkSpectrumSignalParameters> ();
481  txParams->duration = duration;
482  txParams->txPhy = this->GetObject<SpectrumPhy> ();
483  txParams->psd = m_txPsd;
484  txParams->packetBurst = pb;
485  //txParams->ctrlMsgList = ctrlMsgList;
486  txParams->txAntenna = m_antenna;
487  txParams->mcs = mcs;
488  txParams->numSym = numSym;
489  txParams->destinationRnti = destinationRnti;
490  txParams->senderRnti = senderRnti;
491  txParams->size = size;
492  txParams->rbBitmap = rbBitmap;
493 
494  m_channel->StartTx (txParams);
495 
496  // The end of the tranmission is reduced by 1 ns to avoid collision in case of a consecutive tranmission in the same slot.
497  m_endTxEvent = Simulator::Schedule (duration - NanoSeconds(1.0), &MmWaveSidelinkSpectrumPhy::EndTx, this);
498  }
499  break;
500  default:
501  NS_LOG_FUNCTION (this << "Programming Error. Code should not reach this point");
502  }
503  return true;
504 }
